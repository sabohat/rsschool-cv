    width: 100vw;
    overflow-x: hidden;
}

    animation: scroll 30s infinite linear;
    width: fit-content;
}
@keyframes scroll {
    from {
      transform: translateX(0%);
    }
    to {
      transform: translateX(-50%);
    }
}
